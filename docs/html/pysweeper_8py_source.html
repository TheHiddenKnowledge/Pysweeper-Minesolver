100 def resize_map(self, width, height, mines):
101 # Handling string instances of the arguments
102 if isinstance(width, str):
103 width = 0 if width == "" else int(width)
104 if isinstance(height, str):
105 height = 0 if height == "" else int(height)
106 if isinstance(mines, str):
107 mines = 0 if mines == "" else int(mines)
108 # Only resizes map if the width, height, and mine count are within
109 # the allowable ranges
110 if (5 <= width <= 20 and 5 <= height <= 20
111 and mines < width * height):
112 self.width = width
113 self.height = height
114 self.__mines = mines
115 self.reset_map()

119 def __generate_mines(self):
120 mine_counter = 0
121 mine_cells = []
122 # Generate mine random coordinates
123 while mine_counter != self.__mines:
124 mine_cell = [rnd.randint(0, self.width - 1),
125 rnd.randint(0, self.height - 1)]
126 if mine_cell not in mine_cells:
127 mine_cells.append(mine_cell)
128 mine_counter += 1
129 # Adding mines to map
130 for mine_cell in mine_cells:
131 self.__map[mine_cell[1]][mine_cell[0]] = "*"

135 def __generate_numbers(self):
136 for x in range(self.width):
137 for y in range(self.height):
138 if self.__map[y][x] != "*":
139 mine_count = 0
140 # Rotation array for NWES and diagonal combinations
141 rots = [[0, 1], [0, -1], [1, 0], [-1, 0],
142 [1, 1], [-1, -1], [1, -1], [-1, 1]]
143 for rot in rots:
144 x_dir = x + rot[0]
145 y_dir = y + rot[1]
146 if 0 <= x_dir < self.width:
147 if 0 <= y_dir < self.height:
148 # For each rotation add to count if mine
149 if self.__map[y_dir][x_dir] == "*":
150 mine_count += 1
151 # Add mine count to coordinate on map
152 if mine_count > 0:
153 self.__map[y][x] = str(mine_count)

211 def __fill_blanks(self, x, y, fill_list):
212 if [x, y] not in fill_list:

224 if [x_dir, y_dir] not in fill_list:
225 if self.gui_map[y_dir][x_dir]["bg"] == "grey":
226 fill_list.append([x_dir, y_dir])
227 # If rotated coordinate has no number,
228 # perform method on coordinate

235 def __check_grid(self, event):
236 if self.game_status == "running":
237 gui_cell = event.widget
238 x = gui_cell.grid_info()["column"]
239 y = gui_cell.grid_info()["row"]
240 # Only reveal hidden or non-flagged coordinates
241 if gui_cell["bg"] != "white" and gui_cell["text"] != "âš‘":
242 # If coordinate has a number, reveal coordinate
243 if self.__map[y][x] != "*" and self.__map[y][x] != " ":
244 gui_cell.configure(text=self.__map[y][x], bg="white")
245 self.hidden -= 1
246 # If coordinate has no number, fill area
247 elif self.__map[y][x] == " ":
248 fill_list = []
249 self.__fill_blanks(x, y, fill_list)
250 for fill_cord in fill_list:
251 x_dir = fill_cord[0]
252 y_dir = fill_cord[1]
253 self.gui_map[y_dir][x_dir].configure(
254 text=self.__map[y_dir][x_dir], bg="white")
255 self.hidden -= 1
256 # If coordinate is a mine, reveal mine and end game
257 elif self.__map[y][x] == "*":
258 gui_cell.configure(text=self.__map[y][x], bg="red")
259 self.game_status = "lost"
260 self.status_label.configure(text="You lost!")
261 # If all hidden coordinates are revealed, end game
262 if self.hidden == self.__mines:
263 self.game_status = "won"
264 self.status_label.configure(text="You won!")
