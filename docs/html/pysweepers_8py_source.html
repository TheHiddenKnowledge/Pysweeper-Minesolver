97 def resize_map(self, width, height, mines):
98 # Handling string instances of the arguments
99 if isinstance(width, str):
100 width = 0 if width == "" else int(width)
101 if isinstance(height, str):
102 height = 0 if height == "" else int(height)
103 if isinstance(mines, str):
104 mines = 0 if mines == "" else int(mines)
105 # Only resizes map if the width, height, and mine count are within
106 # the allowable ranges
107 if (5 <= width <= 20 and 5 <= height <= 20
108 and mines < width * height):
109 self.width = width
110 self.height = height
111 self.__mines = mines

115 def __generate_mines(self):
116 mine_counter = 0
117 mine_cells = []
118 # Generate mine random coordinates
119 while mine_counter != self.__mines:
120 mine_cell = [rnd.randint(0, self.width - 1),
121 rnd.randint(0, self.height - 1)]
122 if mine_cell not in mine_cells:
123 mine_cells.append(mine_cell)
124 mine_counter += 1
125 # Adding mines to map
126 for mine_cell in mine_cells:
127 self.__map[mine_cell[1]][mine_cell[0]] = "*"

130 def __generate_numbers(self):
131 for x in range(self.width):
132 for y in range(self.height):
133 if self.__map[y][x] != "*":
134 mine_count = 0
135 # Rotation array for NWES and diagonal combinations
136 rots = [[0, 1], [0, -1], [1, 0], [-1, 0],
137 [1, 1], [-1, -1], [1, -1], [-1, 1]]
138 for rot in rots:
139 x_dir = x + rot[0]
140 y_dir = y + rot[1]
141 if 0 <= x_dir < self.width:
142 if 0 <= y_dir < self.height:
143 # For each rotation add to count if mine
144 if self.__map[y_dir][x_dir] == "*":
145 mine_count += 1
146 # Add mine count to coordinate on map
147 if mine_count > 0:
148 self.__map[y][x] = str(mine_count)

226 def __check_grid(self, event):
227 if self.game_status == "running":
228 gui_cell = event.widget
229 x = gui_cell.grid_info()["column"]
230 y = gui_cell.grid_info()["row"]
231 # Only reveal hidden or non-flagged coordinates
232 if gui_cell["bg"] != "white" and gui_cell["text"] != "âš‘":
233 # If coordinate has a number, reveal coordinate
234 if self.__map[y][x] != "*" and self.__map[y][x] != " ":
235 gui_cell.configure(text=self.__map[y][x], bg="white")
236 self.hidden -= 1
237 # If coordinate has no number, fill area
238 elif self.__map[y][x] == " ":
239 fill_list = []
240 self.__fill_blanks(x, y, fill_list)
241 for fill_cord in fill_list:
242 x_dir = fill_cord[0]
243 y_dir = fill_cord[1]
244 self.gui_map[y_dir][x_dir].configure(
245 text=self.__map[y_dir][x_dir], bg="white")
246 self.hidden -= 1
247 # If coordinate is a mine, reveal mine and end game
248 elif self.__map[y][x] == "*":
249 gui_cell.configure(text=self.__map[y][x], bg="red")
250 self.game_status = "lost"
251 self.status_label.configure(text="You lost!")
252 # If all hidden coordinates are revealed, end game
253 if self.hidden == self.__mines:
254 self.game_status = "won"
255 self.status_label.configure(text="You won!")
