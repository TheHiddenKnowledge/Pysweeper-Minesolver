56 mine_counter = 0
57 mine_cells = []
58 # Generate mine random coordinates
59 while mine_counter != self.__mines:
60 mine_cell = [rnd.randint(0, self.width - 1),
61 rnd.randint(0, self.height - 1)]
62 if mine_cell not in mine_cells:
63 mine_cells.append(mine_cell)
